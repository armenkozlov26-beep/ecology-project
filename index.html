async function submitToDatabase() { // Получаем значения по ID const name = document.getElementById('client_name').value; const weight = document.getElementById('weight_kg').value; const wasteId = document.getElementById('waste_type_id').value; if (!name || !weight) { alert("Заполните все поля!"); return; } // Формируем JSON в строгом соответствии с OrderCreate в main.py const data = { client_name: String(name), waste_type_id: parseInt(wasteId), // Обязательно целое число weight_kg: parseFloat(weight) // Обязательно число с запятой }; console.log("Отправка в Render:", data); try { const response = await fetch('https://ecology-project-do42.onrender.com/send', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }); if (response.ok) { alert("✅ Успех! Проверь pgAdmin."); document.getElementById('client_name').value = ""; document.getElementById('weight_kg').value = ""; } else { const errorDetail = await response.json(); console.error("Ошибка 422 детали:", errorDetail); alert("❌ Ошибка 422. Проверь консоль браузера."); } } catch (e) { console.error("Сбой сети:", e); alert("❌ Сервер недоступен."); } }
