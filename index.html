async function sendData() { // Получаем значения по ID элементов const nameVal = document.getElementById('client_name').value; const wasteVal = document.getElementById('waste_type_id').value; const weightVal = document.getElementById('weight_kg').value; if (!nameVal || !weightVal) { alert("Заполните имя и вес!"); return; } // Собираем объект строго по схеме бэкенда const orderData = { client_name: String(nameVal), waste_type_id: parseInt(wasteVal), // Превращаем в целое число (int) weight_kg: parseFloat(weightVal) // Превращаем в число с точкой (float) }; console.log("Отправка данных:", orderData); try { const response = await fetch('https://ecology-project-do42.onrender.com/send', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(orderData) }); if (response.ok) { alert("✅ Данные успешно отправлены в pgAdmin!"); // Очистка формы document.getElementById('client_name').value = ""; document.getElementById('weight_kg').value = ""; } else { const errorText = await response.json(); console.error("Детали ошибки 422:", errorText); alert("❌ Ошибка 422: Сервер отклонил данные. Проверь консоль (F12)."); } } catch (e) { console.error("Ошибка сети:", e); alert("❌ Ошибка соединения с сервером Render."); } }
